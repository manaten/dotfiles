./init.sh
```

# mintty settings(with WSL)

"%LOCALAPPDATA%\wsltty\bin\mintty.exe " --WSL="Ubuntu" --configdir="%APPDATA%\wsltty" --config="%USERPROFILE%\work\github.com\manaten\dotfiles\minttyrc" -~ /usr/bin/zsh -l
```
